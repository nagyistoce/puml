Detailed Description

This is a singleton object that lists all possible nodes. The node objects held in this class are not drawn. Instead, they are kept as a toolbar reference.

void AllNodes::register_node ( BaseNode *  to_register) [inline] 
                                                  {
            debug("> AllNodes::register_node()");
            linked_nodes* cursor;

            if (nodes == NULL) { //Initial registration
                cursor = (linked_nodes*)malloc(sizeof(linked_nodes));
                nodes = cursor;
            } else {
                cursor = nodes;

                while (cursor -> next != NULL) {
                    cursor = cursor -> next;
                }

                cursor -> next = (linked_nodes*)malloc(sizeof(linked_nodes));
                cursor = cursor -> next;
            }

            //cursor now points at last element, where we want to tack on the new node.
            cursor -> node = to_register;
            cursor -> next = NULL;
            debug("< AllNodes::register_node()");
        }
